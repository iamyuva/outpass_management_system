Table Parent {
  parent_id int [pk]
  name varchar(100)
  phone_no varchar(15)
  email varchar(100)
}

Table Student {
  student_id int [pk]
  name varchar(100)
  register_no varchar(20) [unique]
  department varchar(50)
  year int
  section varchar(10)
  phone_no varchar(15)
  email varchar(100)
  parent_id int
}

Table Staff {
  staff_id int [pk]
  name varchar(100)
  department varchar(50)
  email varchar(100)
  phone_no varchar(15)
}

Table Admin {
  admin_id int [pk]
  name varchar(100)
  email varchar(100)
  phone_no varchar(15)
}

Table Caretaker {
  caretaker_id int [pk]
  name varchar(100)
  phone_no varchar(15)
  email varchar(100)
}

Table Approval {
  approval_id int [pk]
  staff_id int
  admin_id int
  caretaker_id int
  approved_by varchar(20) -- values: 'Staff', 'Admin', 'Caretaker'
  approval_date date
  remarks text
}

Table Pass {
  pass_id int [pk]
  student_id int
  type varchar(20) -- values: 'Outpass', 'Homepass'
  reason text
  destination varchar(255)
  date_requested date
  date_of_leaving date
  date_of_return date
  status varchar(20) -- values: 'Pending', 'Approved', 'Rejected'
  approval_id int
}

Table Notification {
  notification_id int [pk]
  pass_id int
  recipient varchar(20) -- values: 'Parent', 'Student'
  message text
  sent_at timestamp
}

Ref: Student.parent_id > Parent.parent_id
Ref: Pass.student_id > Student.student_id
Ref: Pass.approval_id > Approval.approval_id
Ref: Approval.staff_id > Staff.staff_id
Ref: Approval.admin_id > Admin.admin_id
Ref: Approval.caretaker_id > Caretaker.caretaker_id
Ref: Notification.pass_id > Pass.pass_id
